function dbl resultado ( int numero )
dbl as = Calcfact ( numero )

end

function dbl Calcfact ( int num )
if ( num == 0 )
return ( 1 )
end
else
for ( int x = num to num > 1 step x = x - 1 )
int R = R * x
end
end
return ( R )
end
